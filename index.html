🔧 Apps Script 코드

function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetName = (data.keyword||'소싱결과').slice(0,30); var sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName); if (sheet.getLastRow() === 0) { sheet.appendRow(['분석일시','상품명','등급','소싱가(위안)', '소싱가(원)','판매가','실제마진','수요점수','경쟁도','분석','태그']); sheet.getRange(1,1,1,11).setFontWeight('bold'); } var now = new Date().toLocaleString('ko-KR'); (data.products||[]).forEach(function(p){ sheet.appendRow([now,p.name,p.grade,p.sourcingCny||'', p.sourcingPrice,p.sellPrice,p.realMargin||p.margin, p.demandScore,p.competition,p.analysis,(p.tags||[]).join(', ')]); }); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e){ return ContentService.createTextOutput(JSON.stringify({ok:true,msg:'소싱레이더 연결 정상'})) .setMimeType(ContentService.MimeType.JSON); }

📝 설정 방법

  1. 구글 드라이브에서 새 스프레드시트 생성
  2. 확장 프로그램 → Apps Script 열기
  3. 위 코드 붙여넣고 저장 (Ctrl+S)
  4. 배포 → 새 배포 → 유형: 웹 앱 → 액세스: 모든 사용자
  5. 배포 URL을 위에 입력 → 저장
  6. 분석 후 "구글시트" 버튼 클릭!
